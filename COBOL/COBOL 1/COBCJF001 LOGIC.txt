//VARIABLE DEFINITIONS
	//INPUT
	STRING PAINT-REC
		STRING PAINT-EST-NO
		STRING PAINT-DATE
			INTEGER PINT-YY, PAINT-MM, PAINT-DD
		INTEGER PAINT-WALL-SQ-FT, PAINT-DOOR-SQ-FT
		DOUBLE PAINT-PRICE-GAL
	STRING I-DATE
	
	//CONSTANTS
	INTEGER FOOTAGE-UNIT = 115
	INTEGER UNIT-PAINT-GAL = 1
	D0UBLE UNIT-LABOUR = 23.55
	INTEGER RUN-DATE
	
	//CALCULATED
	INTEGER PAGE = 0
	
		//DATA LINE
	INTEGER C-TOT-SQ-FT
	DOUBLE C-GAL-NEED, C-PAINT-EST, C-LABOUR-EST, C-TOT-EST
	
		//GRAND TOTALS
	INTEGER GT-NUM-EST = 0
	DOUBLE GT-GAL-NEED = 0, GT-PAINT-EST = 0, GT-LABOUR-EST = 0, GT-TOT-EST = 0

MAIN
	CALL INIT
	DO UNTIL EOF
		CALL CALCS
		CALL DATA-LINE
		CALL READ
	END-DO
	CALL CLOSING
END-MAIN

INIT
	RUN-DATE = SYSTEM-DATE
	OPEN INPUT FILE (PAINTEST.DAT)
	OPEN OUTPUT FILE (PJOBBEST.PRT)
	CALL HEADER
	CALL READ
END-INIT

CALCS
	TOT-SQ-FT = PAINT-WALL-SQ-FT - PAINT-DOOR-SQ-FT
	C-GAL-NEED = TOT-SQ-FT / FOOTAGE-UNIT * UNIT-PAINT-GAL
	C-PAINT-EST = C-GAL-NEED * PAINT-PRICE-GAL
	C-LABOUR-EST = TOT-SQ-FT / FOOTAGE-UNIT * UNIT-LABOUR
	C-TOT-EST = C-PAINT-EST + C-LABOUR-EST
	
	//GT 
	GT-NUM-EST++
	GT-PAINT-EST += C-PAINT-EST		//paint cost estimate
	GT-LABOUR-EST += C-LABOUR-EST	//labor cost estimate
	GT-TOT-EST += C-TOT-EST
END-CALCS

DATA-LINE
	FORMAT VARIABLES (I'LL JUST USE THE ORIGINAL VARIABLES FOR SIMPLICITIE'S SAKE)
	PRINT DATA LINE: PAINT-EST-NO, PAINT-MM, PAINT-DD, PAINT-YY, PAINT-WALL-SQ-FT, 
		PAINT-DOOR-SQ-FT, C-TOT-EST, C-GAL-NEED, PAINT-PRICE-GAL, C-PAINT-EST, 
		C-LABOUR-EST, C-TOT-EST
	
	CHECK FOR END OF PAGE, IF SO GO TO NEW PAGE AND CALL HEADER
END-DATA-LINE

CLOSING
	FORMAT GRAND TOTAL VARIABLES
	PRINT BLANK LINE
	PRINT GRAND TOTAL LINE: GT-NUM-EST, GT-GAL-NEED, GT-PAINT-EST, GT-LABOUR-EST, 
		GT-TOT-EST
	CLOSE INPUT FILE
	CLOSE OUTPUT FILE
END-CLOSING

HEADER
	PAGE ++
	FORMAT VARIABLES (I'LL JUST USE THE ORIGINAL VARIABLES FOR SIMPLICITIE'S SAKE)
	PRINT COMPANY TITLE LINE
	PRINT BLANK LINE
	PRINT TITLE HEADING LINE 1
	PRINT TITLE HEADING LINE 2
	PRINT BLANK LINE
END-HEADER

READ
	READ RECORD: PAINT-EST-NO, PAINT-YY, PAINT-MM, PAINT-DD, PAINT-WALL-SQ-FT, 
		PAINT-DOOR-SQ-FT, PAINT-PRICE-GAL
	IF (END OF FILE)
		EOF = TRUE
	END-IF
END-READ
