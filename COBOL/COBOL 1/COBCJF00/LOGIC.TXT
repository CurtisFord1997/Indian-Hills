VARIABLE DEFINITIONS
	INPUT
		ALPHANUMERIC - I-REC, I-ID, I-NAME, I-LNAME, I-FNAME, I-INIT
		NEUMERIC - I-GPA, I-START-SALARY, I-DATE 
	CALCULATED
		NEUMERIC - C-SCTR=0, C-PCTR=0
		ALPHANUMERIC -  EOF = "F"
	OUTPUT
		ALPHANUMERIC - O-DATE, O-PAGE, O-ID, O-LNAME, O-FNAME, O-GPA, O-START-SALARY, O-SCTR, O-PCTR

MAIN
	CALL INIT
	WHILE (EOF = "F")
		CALL MAINLINE
	END-WHILE
	CALL CLOSING
END-MAIN

MAINLINE
	CALL CALCS
	CALL OUTPUT
	CALL READ
END-MAINLINE

CALCS
	C-SCTR ++
END-CALCS

OUTPUT
	O-ID = I-ID
	O-LNAME = I-LNAME
	O-FNAME = I-FNAME
	O-GPA = I-GPA
	O-START-SALARY = I-START-SALARY
	PRINT DETAIL LINE
	PRINT BLANK LINE
	IF (END OF PAGE)
		CALL HEADINGS
	END-IF
END-OUTPUT

INIT
	OPEN INNPUT FILE
	OPEN OUTPUT FILE
	REQUEST SYSTEM DATE
	GET I-DATE
	O-DATE = FORMAT(I-DATE)
	CALL READ
	CALL HEADINGS
END-INIT

CLOSING
	PRINT BLANK LINE
	O-SCTR = C-SCTR
	PRINT TOTAL LINE
	CLOSE INPUT FILE
	CLOSE OUTPUT FILE
END-CLOSING

HEADINGS
	O-PCTR = (C-PCTR ++)
	PRINT TITLE LINE
	PRINT BLANK LINE
	PRINT COL HEADING LINE 1
	PRINT COL HEADING LINE 2
	PRINT BLANK LINE
END-HEADINGS

READ
	READ RECORD, I-REC, I-ID, I-NAME, I-LNAME, I-FNAME, I-INIT, I-GPA, I-START-SALARY
	IF END OF FILE
		EOF = "T"
	END-IF
END-READ
